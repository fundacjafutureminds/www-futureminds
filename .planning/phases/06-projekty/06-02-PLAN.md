```tsx
function ProjectCard({ project }: { project: Project }) {
  return (
    <div className="border-l border-[#FFFFFF54] px-[25px] py-6">
      {/* Logo(s) */}
      {project.image && (
        <div className="mb-4 flex items-center gap-4">
          <Image
            src={project.image}
            alt={project.title}
            width={200}
            height={70}
            className="h-auto w-auto max-w-[212px]"
          />
          {project.secondaryImage && (
            <Image
              src={project.secondaryImage}
              alt=""
              width={160}
              height={70}
              className="h-auto w-auto max-w-[159px]"
            />
          )}
        </div>
      )}

      {/* Tytul */}
      <h4 className="mb-4 mr-[45px] text-[22px] font-light leading-[1.1] tracking-[1px] text-[#EFEFEF]">
        {project.href ? (
          <a href={project.href} target="_blank" rel="noopener noreferrer" className="transition-colors hover:text-fm-green">
            {project.title}
          </a>
        ) : (
          project.title
        )}
      </h4>

      {/* Opis */}
      <p className="text-[15px] font-normal leading-[22px] tracking-[0.2px] text-[#E8E8E8]">
        {project.href ? (
          <a href={project.href} target="_blank" rel="noopener noreferrer" className="transition-colors hover:text-fm-green">
            {project.description}
          </a>
        ) : (
          project.description
        )}
      </p>
    </div>
  );
}
```

- NIE uzywac `rounded-sm` — border-radius to 0
- NIE dodawac CTA "WIECEJ" — karty projektow NIE MAJA przyciskow (w przeciwienstwie do kart programow)
- Tytul: 22px, font-light (w300), line-height 1.1, letter-spacing 1px, color #EFEFEF, margin-right 45px
- Opis: 15px, font-normal (w400), line-height 22px, letter-spacing 0.2px, color #E8E8E8
- Logo BEZ filtrow brightness-0 invert — loga projektow sa juz jasne na ciemnym tle
- secondaryImage renderowane obok glownego logo (flex items-center gap-4) — tylko Enigma ma 2 loga
